#include <SPI.h>
#include <MFRC522.h>

#define ledVerde 7
#define ledVermelho 6
#define SS_PIN 10
#define RST_PIN 9
#define pinoFechadura 3
MFRC522 mfrc522(SS_PIN, RST_PIN);   // Criar MFRC522.

char st[20];

void setup()
{
  Serial.begin(9600);   // Inicia a serial
  SPI.begin();      // Inicia  SPI bus
  mfrc522.PCD_Init();   // Inicia MFRC522
  pinMode(pinoFechadura, OUTPUT);
  pinMode(ledVerde, OUTPUT);
  pinMode(ledVermelho, OUTPUT);
  digitalWrite(pinoFechadura, HIGH);
  digitalWrite(ledVerde, LOW);
  digitalWrite(ledVermelho, LOW);
  Serial.println("Aproxime o seu cartao do leitor...");
  Serial.println();
}

void loop()
{
  // Look for new cards
  if ( ! mfrc522.PICC_IsNewCardPresent())
  {
    return;
  }
  // Select one of the cards
  if ( ! mfrc522.PICC_ReadCardSerial())
  {
    return;
  }
  String conteudo = "";
  byte letra;
  for (byte i = 0; i < mfrc522.uid.size; i++)
  {
    conteudo.concat(String(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " "));
    conteudo.concat(String(mfrc522.uid.uidByte[i], HEX));
  }

  conteudo.toUpperCase();
  if (conteudo.substring(1) == "40 BF 4B D3") //UID 1 - Cartao 1
  {
    Serial.println("d"); //cartao denys
    abrirFechadura();
    delay(100);
    conteudo.substring(1) = "";
  }

  else if (conteudo.substring(1) == "40 EE 23 D4") //UID 2 - Cartao 2
  {
    Serial.println("b"); // cartao avila
    abrirFechadura();
    Serial.println();
    delay(100);
  }

  else
  {
    Serial.println("Entrada nao permitida. Favor realizar cadastro."); // qqr outro cartao
    Serial.println("n");
    digitalWrite(ledVermelho, HIGH);
    delay(2000);
    digitalWrite(ledVermelho, LOW);
  }
}

void abrirFechadura() {
  digitalWrite(pinoFechadura, LOW);
  digitalWrite(ledVerde, HIGH);
  delay(5000);
  digitalWrite(pinoFechadura, HIGH);
  digitalWrite(ledVerde, LOW);
}
